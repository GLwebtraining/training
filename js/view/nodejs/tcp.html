TCP

Чтобы использовать основные сетевые функции необходимо подключить встроенный модуль net:
var net = require('net')

net.createServer() - создает новый TCP сервер. Метод возвращает экземпляр созданного сервера и принимает callback функцию - function callback (socket) { /* ... */ }

Чтобы начать работать с созданным сервером нужно вызвать метод listen и передать номер порта:
var net = require('net')
var server = net.createServer(function (socket) {
  /*...*/
});
server.listen(8000);

Объект socket содержит много мета данных, также является Node duplex Stream (может быть использован как для чтения, так и для записи).

socket.write(data) используется для записи данных в сокет
socket.end() - для закрытия сокета. Метод также может принять данные(как и write) и затем закрыть соединение.

var net = require('net');
net.createServer(function(socket){
  socket.write(new Date().toString());
  socket.end('\n');
}).listen(8000);
