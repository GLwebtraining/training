Пространство имен

Пространства имен помогают уменьшить количество глобальных пере менных, необходимых нашим программам, и одновременно избежать конфликтов имен и чрезмерного употребления префиксов.

Вместо того чтобы засорять глобальное пространство имен большим количеством функций, объектов и других переменных, можно создать один (в идеале только один) глобальный объект, который бу дет служить пространством имен для приложения или библиотеки.

Простой пример использования:

var App = {};
App.someMethod = function(){/*...*/};
App.var_a = 1;
App.modules = {};
App.modules.someChildModule = {};

Универсальная функция для создания пространства имен

var App = App || {};
App.namespace = function (ns_string) {
  var parts = ns_string.split('.'),
    parent = MYAPP, i;
  if (parts[0] === “MYAPP”) parts = parts.slice(1);
  for (i = 0; i < parts.length; i++){
    if (typeof parent[parts[i]] === 'undefined') parent[parts[i]] = {};
    parent = parent[parts[i]];
  }
  return parent;
};

Использование:

// присваивать возвращаемое значение локальной переменной
var module2 = App.namespace('App.modules.module2');
App.namespace('modules.module51');
